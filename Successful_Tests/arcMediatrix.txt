mediatriz(p1, p2) = 
let
	l1 = line(p1, p2);
	m = measure (p1, p2);
	c1 = circle (p1, m);
	c2 = circle (p2, m);
	i1,i2,_ = intersect(c1, c2);
	l2 = line(i1, i2);
in l2;
    
p1 = point(250,250);	
draw p1;
p2 = point(250,300);	
draw p2;
p3 = point(300,250);	
draw p3;

med = mediatriz(p2,p3);	
draw med;

m = measure(p1,p2);

a = arc(p1,p3,p2,m);

draw a;

color red;
draw intersect(a,med);
